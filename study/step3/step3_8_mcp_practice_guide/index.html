def resolve_doc_path(filename: str) -> Path:
    \"\"\"
    sample_docs лӮҙл¶Җмқҳ Markdown нҢҢмқј кІҪлЎңл§Ң н—Ҳмҡ©н•ңлӢӨ.
    \"\"\"

    if not str(path).startswith(str(SAMPLE_DOCS_DIR.resolve())):
        raise ValueError("sample_docs л””л үн„°лҰ¬ л°–мқҳ нҢҢмқјмқҖ мқҪмқ„ мҲҳ м—ҶмҠөлӢҲлӢӨ.")

        raise ValueError("Markdown нҢҢмқјл§Ң мқҪмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.")

    if not path.exists():
        raise FileNotFoundError(f"нҢҢмқјмқ„ м°ҫмқ„ мҲҳ м—ҶмҠөлӢҲлӢӨ: {filename}")

    return path

@mcp.tool()
def search_docs(query: str, top_k: int = 3) -> list[dict[str, str]]:
    \"\"\"
    sample_docs Markdown л¬ём„ңм—җм„ң queryк°Җ нҸ¬н•Ёлҗң л¬ём„ңлҘј кІҖмғүн•ңлӢӨ.

    \"\"\"
    if not query or not query.strip():
        raise ValueError("кІҖмғүм–ҙк°Җ л№„м–ҙ мһҲмҠөлӢҲлӢӨ.")

    query_lower = query.lower()
    results: list[dict[str, str]] = []

    for path in sorted(SAMPLE_DOCS_DIR.glob("*.md")):
        content = path.read_text(encoding="utf-8")
        content_lower = content.lower()

        if query_lower in content_lower or query_lower in path.stem.lower():
            preview = content.replace("\\n", " ").strip()

            if len(preview) > 200:
                preview = preview[:200] + "..."

            results.append(
                {
                    "filename": path.name,
                    "title": path.stem,
                    "preview": preview,
                }
            )

    return results[:top_k]


if __name__ == "__main__":
    mcp.run()

def validate_select_query(sql: str) -> None:
    \"\"\"
    SELECT мҝјлҰ¬л§Ң н—Ҳмҡ©н•ңлӢӨ.
    \"\"\"
    normalized = sql.strip().lower()

    if not normalized.startswith("select"):
        raise ValueError("SELECT мҝјлҰ¬л§Ң мӢӨн–үн•  мҲҳ мһҲмҠөлӢҲлӢӨ.")

    blocked_keywords = [
        "insert",
        "update",
        "delete",
        "drop",
        "alter",
        "create",
        "replace",
        "truncate",
    ]

    for keyword in blocked_keywords:
        if keyword in normalized:
            raise ValueError(f"н—Ҳмҡ©лҗҳм§Җ м•ҠмқҖ SQL нӮӨмӣҢл“ңмһ…лӢҲлӢӨ: {keyword}")


@mcp.tool()
def list_tables() -> list[str]:
    \"\"\"
    SQLite DBмқҳ н…Ңмқҙлё” лӘ©лЎқмқ„ мЎ°нҡҢн•ңлӢӨ.
    \"\"\"
    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute(
        "SELECT name FROM sqlite_master WHERE type='table' ORDER BY name"
    )

    rows = cursor.fetchall()
    conn.close()

    return [row["name"] for row in rows]


@mcp.tool()
def describe_table(table_name: str) -> list[dict[str, str]]:
    \"\"\"
    н…Ңмқҙлё” м»¬лҹј м •ліҙлҘј мЎ°нҡҢн•ңлӢӨ.
    \"\"\"
    if not table_name or not table_name.strip():
        raise ValueError("н…Ңмқҙлё”лӘ…мқҙ л№„м–ҙ мһҲмҠөлӢҲлӢӨ.")

    conn = connect_db()
    cursor = conn.cursor()

    cursor.execute(f"PRAGMA table_info({table_name})")

    rows = cursor.fetchall()
    conn.close()

    return [
        {
            "name": row["name"],
            "type": row["type"],
            "not_null": str(bool(row["notnull"])),
            "primary_key": str(bool(row["pk"])),
        }
        for row in rows
    ]


@mcp.tool()
def run_select_query(sql: str, limit: int = 20) -> list[dict]:
    \"\"\"
    SELECT мҝјлҰ¬лҘј мӢӨн–үн•ңлӢӨ.

    \"\"\"

    safe_limit = min(max(limit, 1), 100)

    conn = connect_db()
    cursor = conn.cursor()

    query = f"SELECT * FROM ({sql}) LIMIT {safe_limit}"
    cursor.execute(query)

    rows = cursor.fetchall()
    conn.close()

    return [dict(row) for row in rows]
